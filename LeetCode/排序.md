        int len = intervals.size();
        int i = 0; 
        vector<vector<int>> res;
        while (i < len && intervals[i][1] < newInterval[0]) {
            res.push_back(intervals[i]);
            i++;

            res.push_back(intervals[i++]);

        return res;
