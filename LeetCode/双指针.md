        int left = 0,right = s.size() - 1;
        while (left < right) {
            if (s[left] != s[right])

            left++; right--;
        }
        return true;
    }
    bool isPalindrome(string s,int left,int right){
        while (left < right) {
            if (s[left] != s[right]) return false;
            left++;
            right--;
        }
        return true;
    }
